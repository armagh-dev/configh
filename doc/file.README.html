require 'configh'

module Green include 'configh/configurable'

define_parameter name: 'hue', type: 'populated_string', description: 'common name for this shade of green', required: true, default: 'olive' define_parameter name: 'rgb', type: 'string', description: 'web rgb for this color', required: false define_parameter name: 'web', type: 'boolean', description: 'true for web standard colors', required: true, default: false define_group_validation_callback callback_class: Green, callback_method: :validate_my_green

def Green.validate_my_green( candidate_config )

error = nil
local_custom_hues = %w{ lime neon olive camo }

if candidate_config.web 
  unless candidate_config.green.rgb
    error = "must provide RGB valuees for web colors" 
  end
else
  unless local_custom_hues.include? candidate_config.green.hue
    error = "#{ candidate_config.green.hue } not a recognized local hue"
  end
end

error

end

def Green.do_stuff_with_config( config )

set_color config.green.hue

end

end
